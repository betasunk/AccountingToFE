function debounceSelf(fn, delay) {
  let timer = null;
  return function () {
    let args = arguments,
      _this = this;
    if (timer) {
      timer = null;
      clearTimeout(timer);
    }
    timer = setTimeout(() => {
      fn.apply(_this, args);
    }, delay);
  };
}
